Custom Process Methods

Execute Signature

override global String toString()

returns
Type: String
Must be a JSON-serialized List<Map<String,Object>> representing a collection of multiple parameterized message descriptions.

Usage 

global with sharing class ChatterPlugin {
    
    //marker properties omitted for clarity
    global class ESB {
        global String Description = 'This plugin creates a Chatter FeedItem Post on the user.';
    }
    
    //process logic
    override global String toString() {
        
        FeedItem feedItem = FeedItem(
            Body = 'Hello world',
            ParentId = UserInfo.getUserId()
        );
        
        insert feedItem;
        
        Map<String,Object> parameters = new Map<String,Object>{
            'acme__FeedItemId' => feedItem.Id
        };
        
        List<Map<String,Object>> parametersList = new List<Map<String,Object>>{
            parameters
        };
        
        return Json.serialize(parametersList);
    }
    
}
